const GAS_URL = "https://script.google.com/macros/s/AKfycbykh0r_2E43j4XXYQnu1rvC4NCd-Lypp08UBLgADMe1O2zHESy-c7uICB8EEiEkmXSz9Q/exec";

function switchTab(tab) {
  const entry = document.getElementById("tabEntry");
  const analysis = document.getElementById("tabAnalysis");
  const entryBtn = document.getElementById("tabEntryBtn");
  const analysisBtn = document.getElementById("tabAnalysisBtn");
  if (tab === "entry") {
    entry.classList.remove("hidden"); analysis.classList.add("hidden");
    entryBtn.classList.add("active"); analysisBtn.classList.remove("active");
  } else {
    entry.classList.add("hidden"); analysis.classList.remove("hidden");
    analysisBtn.classList.add("active"); entryBtn.classList.remove("active");
    renderAnalysis();
  }
}

function downloadAnalysisPdf() {
  // 先切到分析分頁確保內容有渲染，再叫出瀏覽器列印，於列印視窗選「另存為 PDF」即可下載
  switchTab("analysis");
  const yf = document.getElementById("yearFilter").value;
  document.getElementById("printTimestamp").textContent = new Date().toLocaleString("zh-TW", { hour12: false });
  document.getElementById("printScope").textContent = yf ? (yf + " 年度") : "全部年度";
  setTimeout(() => window.print(), 150);
}

function jsonpRequest(params) {
  return new Promise((resolve, reject) => {
    const cbName = "cb_" + Date.now() + "_" + Math.floor(Math.random() * 10000);
    window[cbName] = (data) => { resolve(data); cleanup(); };
    const script = document.createElement("script");
    const qs = new URLSearchParams({ ...params, callback: cbName }).toString();
    script.src = GAS_URL + "?" + qs;
    script.onerror = () => { reject(new Error("JSONP 載入失敗")); cleanup(); };
    document.body.appendChild(script);
    function cleanup() {
      delete window[cbName];
      if (script.parentNode) script.parentNode.removeChild(script);
    }
    setTimeout(() => reject(new Error("逾時")), 15000);
  });
}

function loadList() {
  jsonpRequest({ action: "list" }).then((res) => {
    allRecords = (res && res.records) ? res.records : [];
    renderAll();
  }).catch((err) => {
    console.error(err);
    document.getElementById("pivotHead").innerHTML = '<tr><th style="color:#C1685A;">資料載入失敗，請確認 GAS_URL 是否已設定並已部署</th></tr>';
  });
}

function renderAll() {
  populateYearFilter();
  populateProgramSelect();
  renderPivot();
  renderListTable();
  renderSelectionSummary();
  if (!document.getElementById("tabAnalysis").classList.contains("hidden")) {
    renderAnalysis();
  }
}

function populateYearFilter() {
  const sel = document.getElementById("yearFilter");
  const current = sel.value;
  const years = [...new Set(allRecords.map(r => String(r.year)))].sort((a, b) => b - a);
  sel.innerHTML = '<option value="">全部年度</option>' + years.map(y => `<option value="${y}">${y} 年度</option>`).join("");
  sel.value = current && years.includes(current) ? current : "";
}

function populateProgramSelect() {
  const sel = document.getElementById("programSelect");
  const current = sel.value;
  const programs = [...new Set(allRecords.map(r => r.program))].sort();
  sel.innerHTML = '<option value="">請選擇既有計畫，或於右側輸入新計畫</option>' +
    programs.map(p => `<option value="${escapeHtml(p)}">${escapeHtml(p)}</option>`).join("");
  sel.value = current && programs.includes(current) ? current : "";
}

function getFilteredRecords() {
  const yf = document.getElementById("yearFilter").value;
  return yf ? allRecords.filter(r => String(r.year) === yf) : allRecords;
}

function renderPivot() {
  const records = getFilteredRecords();
  const programs = [...new Set(records.map(r => r.program))].sort();
  const head = document.getElementById("pivotHead");
  const body = document.getElementById("pivotBody");

  if (records.length === 0) {
    head.innerHTML = '<tr><th style="font-weight:400; color:#B79E86;">目前尚無資料 🍃</th></tr>';
    body.innerHTML = "";
    return;
  }

  head.innerHTML = `<tr>
      <th>年度</th><th>季度</th>
      ${programs.map(p => `<th class="text-right">${escapeHtml(p)}</th>`).join("")}
      <th class="text-right" style="color:var(--caramel-dark);">🍯 總補助費用</th>
    </tr>`;

  const groups = {};
  records.forEach(r => {
    const key = r.year + "|" + r.quarter;
    if (!groups[key]) groups[key] = { year: r.year, quarter: r.quarter, byProgram: {} };
    groups[key].byProgram[r.program] = (groups[key].byProgram[r.program] || 0) + Number(r.amount);
  });
  const rows = Object.values(groups).sort((a, b) => (b.year - a.year) || (a.quarter - b.quarter));

  const programTotals = {};
  let grandTotal = 0;

  body.innerHTML = rows.map(g => {
    let rowTotal = 0;
    const cells = programs.map(p => {
      const v = g.byProgram[p] || 0;
      rowTotal += v;
      programTotals[p] = (programTotals[p] || 0) + v;
      return `<td class="text-right" style="color:${v ? "var(--brown-text)" : "#C9B6A3"}">${v ? fmtMoney(v) : "—"}</td>`;
    }).join("");
    grandTotal += rowTotal;
    return `<tr>
        <td>${g.year}</td><td>第 ${g.quarter} 季 🍃</td>${cells}
        <td class="text-right font-bold" style="color:var(--caramel-dark);">${fmtMoney(rowTotal)}</td>
      </tr>`;
  }).join("") + `
    <tr class="font-bold" style="background:#FDF4E9; border-top:2px solid var(--line);">
      <td colspan="2">🍡 合　計</td>
      ${programs.map(p => `<td class="text-right">${fmtMoney(programTotals[p] || 0)}</td>`).join("")}
      <td class="text-right" style="color:var(--caramel-dark);">${fmtMoney(grandTotal)}</td>
    </tr>`;
}

function renderListTable() {
  const records = getFilteredRecords().slice().sort((a, b) => (b.year - a.year) || (b.quarter - a.quarter) || (b.createdAt || "").localeCompare(a.createdAt || ""));
  const body = document.getElementById("listBody");
  if (records.length === 0) {
    body.innerHTML = '<tr><td colspan="9" class="text-center py-6" style="color:#B79E86;">尚無登錄紀錄 🍃</td></tr>';
    return;
  }
  body.innerHTML = records.map(r => `
    <tr>
      <td><input type="checkbox" class="rowCheck" data-id="${r.id}" ${selectedIds.has(r.id) ? "checked" : ""} style="width:18px;height:18px;accent-color:var(--caramel-dark);"></td>
      <td>${r.year}</td>
      <td>第 ${r.quarter} 季</td>
      <td>${escapeHtml(r.program)}</td>
      <td class="text-right">${fmtMoney(Number(r.amount))}</td>
      <td style="color:var(--brown-soft);">${escapeHtml(r.note || "")}</td>
      <td>${r.fileUrl ? `<a href="${r.fileUrl}" target="_blank" style="color:var(--caramel-dark); font-weight:700;" class="hover:underline"><i class="fa-solid fa-paperclip mr-1"></i>${escapeHtml(r.fileName || "檔案")}</a>` : '<span style="color:#C9B6A3;">—</span>'}</td>
      <td class="text-xs" style="color:#9C8468;">${escapeHtml(r.createdAt || "")}</td>
      <td class="whitespace-nowrap">
        <button onclick="startEdit('${r.id}')" style="color:var(--caramel-dark);" class="hover:opacity-70 mr-3">✏️</button>
        <button onclick="deleteRecord('${r.id}')" style="color:var(--pink-dark);" class="hover:opacity-70">🗑️</button>
      </td>
    </tr>
  `).join("");

  document.querySelectorAll(".rowCheck").forEach(cb => {
    cb.addEventListener("change", (e) => {
      const id = e.target.getAttribute("data-id");
      if (e.target.checked) selectedIds.add(id); else selectedIds.delete(id);
      renderSelectionSummary();
    });
  });
}

function renderSelectionSummary() {
  const chipsEl = document.getElementById("selectedChips");
  const selected = allRecords.filter(r => selectedIds.has(r.id));
  if (selected.length === 0) {
    chipsEl.innerHTML = '<span class="text-xs" style="color:#B79E86;">目前尚未勾選任何項目</span>';
  } else {
    chipsEl.innerHTML = selected.map(r => `
      <span class="chip">🍬 ${r.year}年 第${r.quarter}季・${escapeHtml(r.program)}：${fmtMoney(Number(r.amount))}
        <button onclick="toggleSelect('${r.id}')" title="移除"><i class="fa-solid fa-circle-xmark"></i></button>
      </span>
    `).join("");
  }
  const total = selected.reduce((sum, r) => sum + Number(r.amount), 0);
  document.getElementById("selCount").textContent = selected.length;
  document.getElementById("selTotal").textContent = fmtMoney(total);
}

function toggleSelect(id) {
  if (selectedIds.has(id)) selectedIds.delete(id); else selectedIds.add(id);
  renderListTable();
  renderSelectionSummary();
}

/* ============ 總覽分析 ============ */
function renderAnalysis() {
  const records = getFilteredRecords();

  const total = records.reduce((s, r) => s + Number(r.amount), 0);
  const programs = [...new Set(records.map(r => r.program))];
  const quarterKeys = new Set(records.map(r => r.year + "|" + r.quarter));
  const avgQuarter = quarterKeys.size ? total / quarterKeys.size : 0;

  document.getElementById("sumTotal").textContent = fmtMoney(total);
  document.getElementById("sumCount").textContent = records.length;
  document.getElementById("sumPrograms").textContent = programs.length;
  document.getElementById("sumAvgQuarter").textContent = fmtMoney(avgQuarter);

  // 各計畫佔比
  const byProgram = {};
  records.forEach(r => { byProgram[r.program] = (byProgram[r.program] || 0) + Number(r.amount); });
  const programEntries = Object.entries(byProgram).sort((a, b) => b[1] - a[1]);
  const maxProgram = programEntries.length ? programEntries[0][1] : 0;
  document.getElementById("programBars").innerHTML = programEntries.length ? programEntries.map(([name, amt]) => {
    const pct = total ? (amt / total * 100) : 0;
    const width = maxProgram ? (amt / maxProgram * 100) : 0;
    return `
      <div>
        <div class="flex justify-between text-sm mb-1">
          <span class="font-bold" style="color:var(--brown-text);">🏷️ ${escapeHtml(name)}</span>
          <span style="color:var(--caramel-dark); font-weight:700;">${fmtMoney(amt)}（${pct.toFixed(1)}%）</span>
        </div>
        <div class="bar-track"><div class="bar-fill" style="width:${width}%;"></div></div>
      </div>`;
  }).join("") : '<p class="text-sm" style="color:#B79E86;">目前尚無資料 🍃</p>';

  // 各年度總額比較
  const byYear = {};
  records.forEach(r => { byYear[r.year] = (byYear[r.year] || 0) + Number(r.amount); });
  const yearEntries = Object.entries(byYear).sort((a, b) => a[0] - b[0]);
  const maxYear = yearEntries.length ? Math.max(...yearEntries.map(e => e[1])) : 0;
  document.getElementById("yearBars").innerHTML = yearEntries.length ? yearEntries.map(([yr, amt]) => {
    const width = maxYear ? (amt / maxYear * 100) : 0;
    return `
      <div>
        <div class="flex justify-between text-sm mb-1">
          <span class="font-bold" style="color:var(--brown-text);">📅 ${yr} 年度</span>
          <span style="color:var(--caramel-dark); font-weight:700;">${fmtMoney(amt)}</span>
        </div>
        <div class="bar-track"><div class="bar-fill" style="width:${width}%;"></div></div>
      </div>`;
  }).join("") : '<p class="text-sm" style="color:#B79E86;">目前尚無資料 🍃</p>';

  // 各季度加總比較（跨年度）
  const byQuarter = { 1: 0, 2: 0, 3: 0, 4: 0 };
  records.forEach(r => { byQuarter[r.quarter] = (byQuarter[r.quarter] || 0) + Number(r.amount); });
  const maxQuarter = Math.max(...Object.values(byQuarter));
  document.getElementById("quarterBars").innerHTML = [1, 2, 3, 4].map(q => {
    const amt = byQuarter[q] || 0;
    const width = maxQuarter ? (amt / maxQuarter * 100) : 0;
    return `
      <div>
        <div class="flex justify-between text-sm mb-1">
          <span class="font-bold" style="color:var(--brown-text);">🍃 第 ${q} 季</span>
          <span style="color:var(--caramel-dark); font-weight:700;">${fmtMoney(amt)}</span>
        </div>
        <div class="bar-track"><div class="bar-fill" style="width:${width}%;"></div></div>
      </div>`;
  }).join("");
}

function deleteRecord(id) {
  if (!confirm("確定要刪除這筆紀錄嗎？此動作無法復原。")) return;
  postToGas({ action: "delete", id }).then(() => {
    allRecords = allRecords.filter(r => r.id !== id);
    selectedIds.delete(id);
    renderAll();
  }).catch(err => alert("刪除失敗：" + err.message));
}

function fileToBase64(file) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onload = () => resolve(reader.result.split(",")[1]);
    reader.onerror = reject;
    reader.readAsDataURL(file);
  });
}

async function onSubmit(e) {
  e.preventDefault();
  const msg = document.getElementById("formMsg");
  const btn = document.getElementById("submitBtn");
  const program = document.getElementById("programCustom").value.trim() || document.getElementById("programSelect").value;

  if (!program) {
    msg.textContent = "請選擇或輸入補助計畫名稱。";
    msg.style.color = "#C1685A";
    return;
  }

  const payload = {
    action: editingId ? "update" : "add",
    id: editingId || ("id_" + Date.now() + "_" + Math.floor(Math.random() * 100000)),
    year: document.getElementById("year").value,
    quarter: document.getElementById("quarter").value,
    program: program,
    amount: document.getElementById("amount").value,
    note: document.getElementById("note").value
  };

  const fileInput = document.getElementById("fileInput");
  if (fileInput.files && fileInput.files[0]) {
    const file = fileInput.files[0];
    if (file.size > 15 * 1024 * 1024) {
      msg.textContent = "檔案過大（上限約 15MB），請壓縮後再上傳。";
      msg.style.color = "#C1685A";
      return;
    }
    payload.fileData = await fileToBase64(file);
    payload.fileName = file.name;
    payload.fileMime = file.type || "application/octet-stream";
  }

  btn.disabled = true;
  msg.textContent = "處理中... 🍯";
  msg.style.color = "var(--brown-soft)";

  postToGas(payload).then(() => {
    msg.textContent = editingId ? "已更新！🎀" : "新增成功！🎉";
    msg.style.color = "#6E8F5C";
    resetForm();
    loadList();
  }).catch(err => {
    msg.textContent = "失敗：" + err.message;
    msg.style.color = "#C1685A";
  }).finally(() => { btn.disabled = false; });
}

// POST 寫入；因 GAS 302 轉址常讀不到回應，失敗時改用 JSONP 依 id 查詢確認是否真的寫入成功
function postToGas(payload) {
  return fetch(GAS_URL, {
    method: "POST",
    headers: { "Content-Type": "text/plain;charset=utf-8" },
    body: JSON.stringify(payload)
  }).then(res => res.json()).then(data => {
    if (data.result === "success") return data;
    throw new Error(data.message || "寫入失敗");
  }).catch(() => {
    if (payload.action === "delete") {
      return { result: "success" };
    }
    return jsonpRequest({ action: "find", id: payload.id }).then(r => {
      if (r.result === "found") return { result: "success" };
      throw new Error("寫入狀態不明，請重新整理頁面確認是否已新增/更新成功");
    });
  });
}

function fmtMoney(n) {
  return "$" + Math.round(n).toLocaleString("zh-TW");
}
function escapeHtml(s) {
  return String(s == null ? "" : s).replace(/[&<>"']/g, c => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;", "'": "&#39;" }[c]));
}
